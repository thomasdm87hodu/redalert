        // --- 3. AUDIO ENGINE ---

        let audioCtx;

        let audioContextEnabled = false;


        function initAudio() {

            if (!audioContextEnabled) {

                audioCtx = new (window.AudioContext || window.webkitAudioContext)();

                if (audioCtx.state === 'suspended') audioCtx.resume();

                audioContextEnabled = true;

            }

        }

        document.addEventListener('click', initAudio, { once: true });


        function playBeep() {

            if (!audioContextEnabled || !audioCtx) return;

            if (audioCtx.state === 'suspended') audioCtx.resume();

            

            const osc = audioCtx.createOscillator();

            const gainNode = audioCtx.createGain();

            

            osc.type = 'square';

            osc.frequency.setValueAtTime(850, audioCtx.currentTime);

            osc.frequency.exponentialRampToValueAtTime(300, audioCtx.currentTime + 0.35);

            

            gainNode.gain.setValueAtTime(0.08, audioCtx.currentTime);

            gainNode.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.35);

            

            osc.connect(gainNode);

            gainNode.connect(audioCtx.destination);

            

            osc.start();

            osc.stop(audioCtx.currentTime + 0.35);

        }


        // --- 4. ALERT LOGIC & UI UPDATES ---

        const activeMarkers = new Map();

        const recentAlerts = new Set(); // Prevent duplicate triggers


        function flashNetworkIndicator() {

            const indicator = document.getElementById('network-indicator');

            indicator.classList.remove('net-activity');

            void indicator.offsetWidth; // trigger reflow

            indicator.classList.add('net-activity');

        }


        function getCoords(cityName) {

            if (cityCoordinates[cityName]) return cityCoordinates[cityName];

            for (let knownCity in cityCoordinates) {

                let mainPart = knownCity.split(' - ')[0].trim();

                if (cityName.includes(mainPart) || knownCity.includes(cityName)) {

                    return cityCoordinates[knownCity];

                }

            }

            // Fallback: Randomize slightly around center of Israel

            return [31.5 + (Math.random() * 1.5 - 0.75), 34.8 + (Math.random() * 0.5 - 0.25)];

        }


        function updateAlertUI(cityName) {

            const list = document.getElementById('alerts-list');

            const noAlerts = document.getElementById('no-alerts-msg');

            if (noAlerts) noAlerts.style.display = 'none';


            const existing = document.getElementById('alert-item-' + cityName);

            if (existing) existing.remove();


            const li = document.createElement('li');

            li.id = 'alert-item-' + cityName;

            li.className = 'bg-red-900 bg-opacity-40 border border-red-500 rounded-md p-3 text-sm text-white flex justify-between items-center animate-flash shadow-md';

            

            const time = new Date().toLocaleTimeString('en-US', { hour12: false });

            li.innerHTML = `<span class="font-bold tracking-wide">${cityName}</span><span class="text-xs text-red-300 font-mono">${time}</span>`;

            list.insertBefore(li, list.firstChild);

        }


        function removeAlertUI(cityName) {

            const existing = document.getElementById('alert-item-' + cityName);

            if (existing) existing.remove();

            

            const list = document.getElementById('alerts-list');

            if (list.children.length === 0 || (list.children.length === 1 && list.children[0].id === 'no-alerts-msg')) {

                const noAlerts = document.getElementById('no-alerts-msg');

                if (noAlerts) noAlerts.style.display = 'block';

            }

        }


        function triggerAlert(cityName) {

            if (activeMarkers.has(cityName)) {

                clearTimeout(activeMarkers.get(cityName).timeout);

                activeMarkers.get(cityName).marker.remove();

                activeMarkers.delete(cityName);

            }

            

            const coords = getCoords(cityName);

            const icon = L.divIcon({

                className: 'radar-blip-container',

                html: `<div class="radar-pulse-1"></div><div class="radar-pulse-2"></div><div class="radar-blip"></div>`,

                iconSize: [24, 24]

            });


            const marker = L.marker(coords, { icon }).addTo(map);

            map.flyTo(coords, Math.max(map.getZoom(), 10), { duration: 1.5 });


            updateAlertUI(cityName);

            playBeep();


            const timeout = setTimeout(() => {

                marker.remove();

                activeMarkers.delete(cityName);

                removeAlertUI(cityName);

            }, 45000); // clear after 45 seconds


            activeMarkers.set(cityName, { marker, timeout });

        }


        function simulateAlert() {

            initAudio(); 

            const cities = Object.keys(cityCoordinates);

            const randomCity = cities[Math.floor(Math.random() * cities.length)];

            processIncomingData([randomCity]);

        }


        // Central payload processor

        function processIncomingData(citiesArray) {

            if (!citiesArray || !Array.isArray(citiesArray) || citiesArray.length === 0) return;

            flashNetworkIndicator();

            

            citiesArray.forEach(city => {

                let cityName = typeof city === 'string' ? city : (city.name || city.zone || "");

                if (cityName && !recentAlerts.has(cityName)) {

                    recentAlerts.add(cityName);

                    triggerAlert(cityName);

                    // Prevent duplicate processing of the exact same city for 60 seconds

                    setTimeout(() => recentAlerts.delete(cityName), 60000); 

                }

            });

        }

        // Official Home Front Command (Oref) via Custom Vercel Proxy

        async function pollOrefAPI() {

            const statusEl = document.getElementById('oref-status');

            try {

                // Call our own dedicated Vercel serverless backend

                const response = await fetch('/api/proxy', { 

                    method: 'GET',

                    cache: 'no-store' 

                });


                if (response.ok) {

                    const textData = await response.text();

                    

                    // Oref returns empty string when there are no alerts

                    if (textData && textData.trim().length > 0) {

                        try {

                            const data = JSON.parse(textData);

                            if (data && data.data && Array.isArray(data.data)) {

                                processIncomingData(data.data);

                            }

                        } catch(e) {

                            // Ignored: Data isn't JSON or is malformed html

                        }

                    }

                    statusEl.innerHTML = 'OK (Live)';

                    statusEl.className = 'text-green-400 font-mono text-xs';

                } else {

                    throw new Error(`HTTP ${response.status}: ${response.statusText}`);

                }

            } catch (err) {

                const time = new Date().toLocaleTimeString('en-US', { hour12: false });

                lastErrorDetails = `[${time}] Fetch Error:\n${err.message}\n\nOur custom Vercel backend had trouble reaching Oref. This usually resolves automatically in a few seconds.\n\nStatus: Retrying...`;

                

                statusEl.innerHTML = `Error (Retrying) <button onclick="showDebug()" class="text-blue-400 underline hover:text-blue-300 ml-1 cursor-pointer">Debug</button>`;

                statusEl.className = 'text-red-400 font-mono text-xs';

            }

        }


        // Run Polling Engine

        setInterval(pollOrefAPI, 2500); // Check API every 2.5 seconds


        // Initial Ping

        pollOrefAPI();
